set (SOURCES add.c test.c)

if (${TRALLOC_SHARED})
    add_library (${TRALLOC_TARGET}-libtest-threads-basic SHARED ${SOURCES})
    target_link_libraries (${TRALLOC_TARGET}-libtest-threads-basic ${TRALLOC_TARGET})
endif ()
if (${TRALLOC_STATIC})
    add_library (${TRALLOC_TARGET}-libtest-threads-basic-static STATIC ${SOURCES})
    target_link_libraries (${TRALLOC_TARGET}-libtest-threads-basic-static ${TRALLOC_TARGET}-static)
    set_target_properties (${TRALLOC_TARGET}-libtest-threads-basic-static PROPERTIES OUTPUT_NAME ${TRALLOC_TARGET}-libtest-threads-basic-static)
endif ()

if (${TRALLOC_SHARED})
    set (NAME ${TRALLOC_TARGET}-test-threads-basic)
    add_executable (${NAME} main.c)
    target_link_libraries (${NAME} ${TRALLOC_TARGET}-libtest-threads-basic)
    add_test (${NAME} ${NAME})
endif ()
if (${TRALLOC_STATIC})
    set (NAME ${TRALLOC_TARGET}-test-threads-basic-static)
    add_executable (${NAME} main.c)
    target_link_libraries (${NAME} ${TRALLOC_TARGET}-libtest-threads-basic-static)
    add_test (${NAME} ${NAME})
endif ()
